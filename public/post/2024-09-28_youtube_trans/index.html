# Este script sirve para descargar las transcripciones de todos los videos de

import os
from pytubefix import YouTube, Playlist
from llama_index.readers.youtube_transcript import YoutubeTranscriptReader

# Funci√≥n para obtener el t√≠tulo del video de YouTube
def obtener_titulo_youtube(url):
    try:
        video = YouTube(url)
        return video.title
    except Exception as e:
        print(f"Error al obtener el t√≠tulo del video {url}: {e}")
        return None

# Funci√≥n para obtener la transcripci√≥n del video de YouTube
def obtener_transcripcion_youtube(url, language='es'):
    try:
        # Usar YoutubeTranscriptReader para obtener la transcripci√≥n
        loader = YoutubeTranscriptReader()
        transcript = loader.load_data(ytlinks=[url], languages=[language])
        return transcript
    except Exception as e:
        print(f"Error al obtener la transcripci√≥n del video {url}: {e}")
        return None

# Funci√≥n para guardar la transcripci√≥n en un archivo Markdown dentro del subdirectorio de la lista
def guardar_transcripcion_en_markdown(titulo, transcripcion, subdirectorio):
    # Crear un nombre de archivo basado en el t√≠tulo, eliminando caracteres no v√°lidos
    nombre_archivo = f"{titulo}.md".replace("/", "-").replace("\\", "-")
    
    # Crear el contenido en formato Markdown
    contenido_md = f"# Transcripci√≥n del video: {titulo}\n\n"
    
    # Iterar sobre los objetos Document y extraer el texto
    for doc in transcripcion:
        contenido_md += f"- {doc.text}\n"
    
    # Crear el subdirectorio si no existe
    if not os.path.exists(subdirectorio):
        os.makedirs(subdirectorio)
    
    # Guardar el archivo en el subdirectorio
    ruta_archivo = os.path.join(subdirectorio, nombre_archivo)
    with open(ruta_archivo, "w", encoding="utf-8") as file:
        file.write(contenido_md)
    
    print(f"Transcripci√≥n guardada en {ruta_archivo}")

# Funci√≥n para imprimir la estructura de la transcripci√≥n
def mostrar_componentes_transcripcion(transcripcion):
    print("\n--- Componentes de la transcripci√≥n ---\n")
    for idx, doc in enumerate(transcripcion):
        print(f"Documento {idx + 1}:")
        print(f"Texto: {doc.text}")
        print(f"Metadatos: {doc.metadata}")
        print("---------------------------\n")

# Funci√≥n para procesar un video de YouTube
def procesar_video_youtube(url, subdirectorio):
    try:
        # Obtener el t√≠tulo del video
        titulo = obtener_titulo_youtube(url)
        if titulo:
            print(f"T√≠tulo del video: {titulo}")
            
            # Obtener la transcripci√≥n del video
            transcripcion = obtener_transcripcion_youtube(url)
            if transcripcion:
                # Mostrar componentes de la transcripci√≥n
                mostrar_componentes_transcripcion(transcripcion)
                
                # Guardar la transcripci√≥n en un archivo Markdown dentro del subdirectorio
                guardar_transcripcion_en_markdown(titulo, transcripcion, subdirectorio)
            else:
                print(f"No se pudo obtener la transcripci√≥n del video: {url}")
        else:
            print(f"URL de video no v√°lida o no se pudo obtener el t√≠tulo para {url}.")
    except Exception as e:
        print(f"Error al procesar el video {url}: {e}")

# Funci√≥n para crear un subdirectorio √∫nico para la lista de reproducci√≥n
def crear_subdirectorio_lista(nombre_base="lista"):
    contador = 1
    subdirectorio = nombre_base
    
    # Verificar si el subdirectorio ya existe y, si es as√≠, crear uno numerado
    while os.path.exists(subdirectorio):
        subdirectorio = f"{nombre_base}{contador:02d}"
        contador += 1
    
    os.makedirs(subdirectorio)
    return subdirectorio

# Funci√≥n para procesar una lista de reproducci√≥n de YouTube usando Playlist de pytubefix
def procesar_lista_reproduccion(url_playlist):
    try:
        # Cargar la lista de reproducci√≥n
        p = Playlist(url_playlist)
        
        # Crear un subdirectorio para la lista de reproducci√≥n
        subdirectorio_lista = crear_subdirectorio_lista("lista")
        print(f"Guardando transcripciones en el directorio: {subdirectorio_lista}")
        
        # Recorrer todas las URLs de videos en la lista de reproducci√≥n
        for url in p.video_urls:
            try:
                print(f"\nProcesando video: {url}")
                
                # Procesar el video y guardar la transcripci√≥n en el subdirectorio de la lista
                procesar_video_youtube(url, subdirectorio_lista)
            except Exception as e:
                # Manejo de errores en cada video
                print(f"Error al procesar el video {url}. Continuando con el siguiente video: {e}")
    except Exception as e:
        print(f"Error al procesar la lista de reproducci√≥n: {e}")


# Ejemplo de uso con una lista de reproducci√≥n de YouTube
